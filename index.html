Simple.Box

Really easy, nice syntax

A regular modal appearing mid screen

var box = Simple.Box.Builder()
.setHeader("Awesome!")
.setContent("My first modal - all teh contents")
.addButton('Ok, I get it', 'btn-success')
.build();

Auto-resizing modal with a minimum size limit

var box = Simple.Box.Builder()
.setHeader("Resize!")
.setContent(content)
.addButton('close')
.setSetting('autoResize', true)
.setSetting('size', {min:400})
.build();

Multiple buttons, multiple callbacks

var onAgree = function()
{
alert('You have agreed to everything');
};
var onDisagree = function()
{
alert('You agree to nothing I guess');
};
var box = Simple.Box.Builder()
.setHeader("Many buttons!")
.setContent("Sir, you have lots of choices")
.addButton('Agree', 'btn-success', onAgree)
.addButton('Disagree', 'btn-danger', onDisagree)
.build();

Unclosable

var onClick = function()
{
alert('Will not close!');
return false;
};
var box = Simple.Box.Builder()
.setHeader("Un-closable")
.setContent("Try if you can")
.addButton('Try to close', 'btn-primary', onClick)
.build();

Shorthand syntax

var box = new Simple.Box("RLY SHRT", "Shorthand setup", [{label:"Ok"}]).render();

autoPosition X axis

var box = new Simple.Box()
.setContent('Im centered horizontally!')
.setSetting('closeOnEscape', true)
.setSetting('autoPositionX', true)
.render();
